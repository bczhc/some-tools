131      /**
132       * get dest file name
133       *
134       * @param file source file
135       * @param dT   0: qmc0/qmcflac
136       *             1: kwm
137       *             2[0-9]{1} : Base128 {
138       *             21: Base128 encode
139       *             22: Base128 decode
140       *             }
141       * @return dest file
142       */
143      @SuppressWarnings("SpellCheckingInspection")
144      private String x(File file, int dT) {
145          String name = file.getName();
146          String name_no_x, x = "";
147          if (name.matches(".*\\..*")) {
148              int index = name.lastIndexOf('.');
149              name_no_x = name.substring(0, index);
150              x = name.substring(index + 1).toLowerCase();
151          } else {
152              name_no_x = name;
153          }
154          String p = file.getParent();
155          switch (dT) {
156              case 0:
157                  try {
158                      switch (x) {
159                          case "qmc0":
160                              return p + "/" + name_no_x + ".mp3";
161                          case "qmcflac":
162                              return p + "/" + name_no_x + ".flac";
163                      }
164                  } catch (StringIndexOutOfBoundsException ignored) {
165                  }
166                  break;
167              case 1:
168                  if (!(x.equals("kwm") | x.equals("kwd"))) return null;
169                  String r;
170                  if (name.matches(".*\\..*")) {
171                      r = p + "/" + name_no_x + ".flac";
172                  } else {
173                      r = p + "/" + name + ".flac";
174                  }
175                  return r;
176              case 21:
177                  return p + "/" + name + ".base128e";
178              case 22:
179                  return p + "/" + name + ".base128d";
180          }
181          return null;
182      }

279       * @param o       decode 1

281       * @param dT      o:1 be ignored

288      private View.OnClickListener getV(int o, int dT, @Nullable Button[] buttons) {
289          return v -> {
290              allButtonsAction(o, buttons, INVISIBLE);
291              if (isDecoding) {
292                  makeText(this, R.string.have_task, LENGTH_SHORT).show();
293                  return;
294              }
295              if (isFolder) {
296                  File[] files = folder.listFiles();
297                  try {
298                      new Thread(() -> {
299                          try {
300                              isDecoding = true;
301                              int i = 1;
302                              int length = files.length;
303                              for (File file : files) {
304                                  if (file.isFile()) {
305                                      try {
306                                          int finalI = i;
307                                          runOnUiThread(() -> mainTv.setText(String.format(getResources().getString(R.string.tv), finalI + " of " + length + ": " + file.getName())));
308                                          String x = x(file, o == 1 ? this.dT : dT);
309                                          if (x != null) {
310                                              String fPath = file.getCanonicalPath();
311                                              if (file.length() != 0L) {
312                                                  int status = new Main().JNI_Decode(fPath, x, o == 1 ? this.dT : dT, tv, MainActivity.this);
313                                                  if (status == -1)
314                                                      runOnUiThread(() -> makeText(this, R.string.fopen_error, LENGTH_SHORT).show());
315                                                  String md5 = MD5Class.conVertFileToMD5(x);
316                                                  runOnUiThread(() -> makeText(this, md5, LENGTH_SHORT).show());
317                                              }
318                                          }
319                                      } catch (IOException e) {
320                                          picker_o.showException(e, MainActivity.this);
321                                          reset();
322                                          allButtonsAction(o, buttons, VISIBLE);
323                                      }
324                                  }
325                                  ++i;
326                              }
327                              runOnUiThread(() -> {
328                                  tv.setText(R.string.percent_100);
329                                  makeText(this, o == 1 ? R.string.all_file_decoded_done : (dT == 21 ? R.string.all_file_encoded_done : R.string.all_file_decoded_done), LENGTH_SHORT).show();
330                                  this.folder = null;
331                                  this.f = null;
332                                  allButtonsAction(o, buttons, VISIBLE);
333                                  mainTv.setText(R.string.nul);
334                                  tv.setText(R.string.nul);
335                                  reset();
336                              });
337                          } catch (Exception e) {
338                              picker_o.showException(e, MainActivity.this);
339                              runOnUiThread(() -> {
340                                  makeText(this, e.toString(), LENGTH_SHORT).show();
341                                  dB.setVisibility(VISIBLE);
342                                  reset();
343                              });
344                          }
345                      }).start();
346                      reset();
347                  } catch (Exception e) {
348                      picker_o.showException(e, MainActivity.this);
349                      reset();
350                      allButtonsAction(o, buttons, VISIBLE);
351                  }
352              } else {
353                  String dF = null;
354                  try {
355                      dF = x(new File(f), o == 1 ? this.dT : dT);
356  //                }
357                  } catch (Exception e) {
358                      e.printStackTrace();
359                      if (!(e instanceof NullPointerException))
360                          makeText(this, e.toString(), LENGTH_SHORT).show();
361                      reset();
362                      allButtonsAction(o, buttons, VISIBLE);
363                  }
364                  try {
365                      if (dF == null) {
366                          runOnUiThread(() -> {
367                              if (toasting != null) {
368                                  toasting.cancel();
369                              }
370                              toasting = makeText(this, R.string.incorrect_file_extension, LENGTH_SHORT);
371                              toasting.show();
372                              allButtonsAction(o, buttons, VISIBLE);
373                          });
374                          return;
375                      }
376                      String finalDF = dF;
377                      new Thread(() -> {
378                          boolean size0 = false;
379                          try {
380                              isDecoding = true;
381                              try {
382                                  size0 = new File(f).length() == 0L;
383                                  int status = new Main().JNI_Decode(f, finalDF, o == 1 ? this.dT : dT, tv, MainActivity.this);
384                                  if (!size0 && (status == -1 || status == 255)) {
385                                      runOnUiThread(() -> makeText(this, R.string.fopen_error, LENGTH_SHORT).show());
386                                  }
387                                  if (status == 1) {
388                                      runOnUiThread(() -> makeText(this, R.string.native_error, LENGTH_SHORT).show());
389                                  }
390                                  String md5 = MD5Class.conVertFileToMD5(finalDF);
391                                  runOnUiThread(() -> makeText(this, md5, LENGTH_SHORT).show());
392                              } catch (Exception e) {
393                                  picker_o.showException(e, MainActivity.this);
394                                  reset();
395                                  allButtonsAction(o, buttons, VISIBLE);
396                              }
397                              isDecoding = false;
398                              if (!size0 && new File(finalDF).exists() && finalDF.length() > 0) {
399                                  runOnUiThread(() -> {
400                                      tv.setText(R.string.percent_100);
401                                      makeText(this, o == 1 ? R.string.decode_done : (dT == 21 ? R.string.encode_done : R.string.decode_done), LENGTH_SHORT).show();
402                                      reset();
403                                  });
404                              }
405                              if (size0)
406                                  runOnUiThread(() -> makeText(this, R.string.null_file, LENGTH_SHORT).show());
407                              allButtonsAction(o, buttons, VISIBLE);
408                          } catch (Exception e) {
409                              picker_o.showException(e, MainActivity.this);
410                              reset();
411                              allButtonsAction(o, buttons, VISIBLE);
412                          }
413                      }).start();
414                  } catch (Exception e) {
415                      picker_o.showException(e, MainActivity.this);
416                      reset();
417                      allButtonsAction(o, buttons, VISIBLE);
418                  }
419              }
420          };
421      }
